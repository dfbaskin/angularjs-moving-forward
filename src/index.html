<div ng-controller="exampleCtrl as vm">
 <div>
  <input type="text"
    ng-model="vm.newItem"
    ng-keypress="($event.which === 13) &amp;&amp; vm.addItem(vm.newItem)" />
  <button type="button"
    ng-click="vm.addItem(vm.newItem)">Add</button>
  </div>
  <div>
    <ol>
      <li ng-repeat="item in vm.items">
        <button type="button"
          ng-click="vm.removeItem(item)">x</button>
          {{item}}
        </li>
    </ol>
  </div>
</div>
class ExampleCtrl {

  constructor() {
    this.items = ["One", "Two", "Three"];
    this.newItem = "Four";
  }

  addItem(item) {
    this.items.push(item);
    this.newItem = "";
  }

  removeItem(item) {
    var idx = this.items.indexOf(item);
    if(idx !== -1) {
      this.items.splice(idx, 1);
    }
  }
}
